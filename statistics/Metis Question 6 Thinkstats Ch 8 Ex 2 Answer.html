def SimulateSample_ex8(lam=2, n=2, m=1000):
    means = []
    for j in range(m):
        xs = np.random.exponential(lam, n)
        xbar = np.mean(xs)
        means.append(xbar)

    cdf = thinkstats2.Cdf(means)
    ci = cdf.Percentile(5), cdf.Percentile(95)
    stderr = RMSE(means, lam)
    return cdf, ci, stderr
